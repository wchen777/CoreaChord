14    /**

28    /**
29     * Begins the REPL.
30     */
31    public void run() {
32      String line = "";
33      try (BufferedReader reader = new BufferedReader(new InputStreamReader(System.in))) {
34        while ((line = reader.readLine()) != null) {
35          String[] input = line.split(" ");
36          Executable app = commands.get(input[0]);
37          if (app != null) {
38            app.execute(input);
39          } else {
40            System.out.println("ERROR: Command DNE :/");
41          }
42        }
43      } catch (IOException e) {
44        System.out.println("ERROR: Buffered Reader error");
45      }
46    }
